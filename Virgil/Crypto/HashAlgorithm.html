def self.convert_to_native(algorithm)
  if ALGORITHMS_TO_NATIVE.has_key?(algorithm)
    return ALGORITHMS_TO_NATIVE[algorithm]
  end

  raise UnknownAlgorithmException("KeyPairType not found: #{algorithm}")
end
